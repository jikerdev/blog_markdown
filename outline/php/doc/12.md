# PHP5.6编译命令

PHP 5.6.5 编译配置命令如下

```bash
./configure \
--prefix=/usr/local/php-5.6.5 \
--with-config-file-path=/usr/local/php-5.6.5/etc \
--enable-fpm \
--with-mysql=mysqlnd \
--with-mysqli=mysqlnd \
--with-pdo-mysql=mysqlnd \
--with-readline \
--enable-mbstring \
--with-gd \
--enable-gd-native-ttf \
--with-jpeg-dir \
--with-png-dir \
--with-freetype-dir=/usr/local/freetype-2.3.5 \
--with-libxml-dir \
--with-pcre-dir \
--with-zlib \
--with-curl \
--enable-sockets \
--with-mcrypt \
--enable-ftp \
--disable-ipv6 \
--with-mhash \
--with-iconv-dir \
--with-bz2 \
--with-gettext \
--without-pear  \
--disable-phar \
--without-pear  \
--disable-phar \
--enable-soap
```
